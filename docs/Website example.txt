javascript// parallax.js
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

gsap.registerPlugin(ScrollTrigger);

export function initParallax() {
  // Check for reduced motion preference
  if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
    return;
  }

  // ===================
  // PARALLAX LAYERS
  // ===================
  const parallaxLayers = document.querySelectorAll('[data-speed]');
  
  parallaxLayers.forEach(layer => {
    const speed = parseFloat(layer.dataset.speed) || 0.5;
    const direction = layer.dataset.direction || 'vertical';
    
    gsap.to(layer, {
      y: direction === 'vertical' ? () => layer.offsetHeight * speed * -1 : 0,
      x: direction === 'horizontal' ? () => layer.offsetWidth * speed * -1 : 0,
      ease: 'none',
      scrollTrigger: {
        trigger: layer.parentElement,
        start: 'top bottom',
        end: 'bottom top',
        scrub: true,
        invalidateOnRefresh: true
      }
    });
  });

  // ===================
  // SECTION HERO PARALLAX
  // ===================
  const sectionHeroes = document.querySelectorAll('.section-hero');
  
  sectionHeroes.forEach(hero => {
    const bg = hero.querySelector('.parallax-bg');
    const mid = hero.querySelector('.parallax-mid');
    const front = hero.querySelector('.parallax-front');
    const content = hero.querySelector('.section-hero-content');
    
    const tl = gsap.timeline({
      scrollTrigger: {
        trigger: hero,
        start: 'top top',
        end: 'bottom top',
        scrub: 1,
        pin: false
      }
    });
    
    if (bg) tl.to(bg, { y: '30%', ease: 'none' }, 0);
    if (mid) tl.to(mid, { y: '50%', ease: 'none' }, 0);
    if (front) tl.to(front, { y: '70%', ease: 'none' }, 0);
    if (content) tl.to(content, { y: '100%', opacity: 0, ease: 'none' }, 0);
  });

  // ===================
  // 3D CARD HOVER EFFECT
  // ===================
  const cards3D = document.querySelectorAll('[data-3d-hover]');
  
  cards3D.forEach(card => {
    card.addEventListener('mousemove', (e) => {
      const rect = card.getBoundingClientRect();
      const x = e.clientX - rect.left;
      const y = e.clientY - rect.top;
      const centerX = rect.width / 2;
      const centerY = rect.height / 2;
      const rotateX = (y - centerY) / 10;
      const rotateY = (centerX - x) / 10;
      
      card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
    });
    
    card.addEventListener('mouseleave', () => {
      card.style.transform = 'perspective(1000px) rotateX(0) rotateY(0)';
    });
  });
}

javascript// navigation.js
import gsap from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

gsap.registerPlugin(ScrollTrigger);

export function initNavigation() {
  const sidebar = document.querySelector('[data-sidebar]');
  const header = document.querySelector('[data-header]');
  const sections = document.querySelectorAll('[data-section-theme]');
  const navLinks = document.querySelectorAll('.section-list a');
  
  // ===================
  // ACTIVE SECTION TRACKING
  // ===================
  sections.forEach(section => {
    ScrollTrigger.create({
      trigger: section,
      start: 'top center',
      end: 'bottom center',
      onEnter: () => updateActiveSection(section),
      onEnterBack: () => updateActiveSection(section)
    });
  });
  
  function updateActiveSection(section) {
    const sectionId = section.id;
    const theme = section.dataset.sectionTheme;
    
    // Update nav active state
    navLinks.forEach(link => {
      link.classList.toggle('active', link.getAttribute('href') === `#${sectionId}`);
    });
    
    // Update sidebar/header theme
    sidebar.dataset.theme = theme;
    header.dataset.theme = theme;
    
    // Update sidebar artwork
    updateSidebarArtwork(sectionId);
  }
  
  // ===================
  // SIDEBAR ARTWORK
  // ===================
  function updateSidebarArtwork(sectionId) {
    const artworkContainer = document.querySelector('[data-sidebar-art]');
    if (!artworkContainer) return;
    
    // Fade transition
    gsap.to(artworkContainer, {
      opacity: 0,
      duration: 0.3,
      onComplete: () => {
        // Update image source based on section
        const img = artworkContainer.querySelector('img') || document.createElement('img');
        img.src = `assets/images/sidebar/${sectionId}.webp`;
        if (!artworkContainer.contains(img)) {
          artworkContainer.appendChild(img);
        }
        gsap.to(artworkContainer, { opacity: 0.3, duration: 0.3 });
      }
    });
  }
  
  // ===================
  // HEADER SCROLL EFFECT
  // ===================
  ScrollTrigger.create({
    start: 'top -100',
    onUpdate: (self) => {
      header.dataset.scrolled = self.progress > 0;
    }
  });
  
  // ===================
  // SMOOTH SCROLL LINKS
  // ===================
  navLinks.forEach(link => {
    link.addEventListener('click', (e) => {
      e.preventDefault();
      const target = document.querySelector(link.getAttribute('href'));
      if (target) {
        gsap.to(window, {
          duration: 1,
          scrollTo: { y: target, offsetY: 0 },
          ease: 'power2.inOut'
        });
      }
    });
  });
}
